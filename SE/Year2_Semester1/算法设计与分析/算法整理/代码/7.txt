//循环递归打印m行n星号

#include <iostream>
using namespace std;

void loopprint(int n,int m)//自己写的循环
{
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
            cout<<"*"<<" ";
        cout<<endl;
    }
}

void fprint(int n,int m)//递归
{
    if (n==0) return ;
    else fprint(n-1,m);
    for (int j=1; j<=m; j++)
        cout<<"*"<<" ";
    cout<<endl;
}
int main()
{
    int n=8,m=5;
    fprint(n,m);
    //cout<<endl;
    //loopprint(n,m);
    return 0;
}
